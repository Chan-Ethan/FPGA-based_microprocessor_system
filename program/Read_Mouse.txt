// Read Mouse data repeatly and display it on Seven Segment Display
@00 00 // Read mouse status from memory to A
@01 A0
@02 02 // Write mouse status to LEDs
@03 C0
@04 00 // Read mouse X position
@05 A1
@06 01 // Read mouse Y position
@07 A2
@08 02 // Write mouse X position to Seg7[3:2]
@09 D0
@0A 03 // Write mouse Y position to Seg7[1:0]
@0B D1
@0C 08 // Jump to IDLE and wait for interrupts

// Read Slide Switch data
@0D 00 // Read slide switch data [15:8] to A
@0E E0
@0F 01 // Read slide switch data [7:0] to B
@10 E1
@0C 08 // Jump to IDLE and wait for interrupts
